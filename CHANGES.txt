Deploy now:
  cd /opt/prompt-tips
  sudo -u www-data npm install --omit=dev
  sudo -u www-data mkdir -p /var/lib/prompt-tips/data/seeds
  sudo cp -f data/seeds/*.yaml /var/lib/prompt-tips/data/seeds/ || true
  sudo -u www-data DATA_DIR=/var/lib/prompt-tips/data npm run seed
  sudo -u www-data DATA_DIR=/var/lib/prompt-tips/data npm run rebuild
  sudo systemctl restart prompt-tips
